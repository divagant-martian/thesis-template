% inserting figures
\newcommand{\insertfigure}[4]{ % Filename, Caption, Label, Width % of textwidth
    \begin{figure}[htbp]
        \begin{center}
            \includegraphics[width=#4\textwidth]{#1}
        \end{center}
        \vspace{-0.4cm}
        \caption{#2}
        \label{#3}
    \end{figure}
}

\newcommand{\inserttable}[4]{ % Caption, tabular params, the table
  \begin{table}[hbp]
    \begin{center}
      \begin{tabular}{#2}
        \toprule
        #3
        \bottomrule
      \end{tabular}\caption{#1}\label{#4}\par
    \end{center}
  \end{table}
}
